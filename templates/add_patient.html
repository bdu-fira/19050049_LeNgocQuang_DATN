Cấu hình ESP32

Để thiết bị ESP32 hoạt động, cần cấu hình:

  • WiFi connection
  • Server endpoint: /api/sensor_data
  • Device ID phải khớp với mã nhập
  • Gửi dữ liệu định kỳ (30s)
Cấu hình ESP32

Sử dụng mã Arduino sau để cấu hình ESP32:

#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>

const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";
const char* serverURL = "http://YOUR_SERVER_IP:5000/api/sensor_data";
const char* deviceID = "ESP32_DEVICE_ID"; // Thay bằng mã thiết bị

void setup() {
    Serial.begin(115200);
    WiFi.begin(ssid, password);
    
    while (WiFi.status() != WL_CONNECTED) {
        delay(1000);
        Serial.println("Connecting to WiFi...");
    }
    Serial.println("Connected to WiFi");
}

void loop() {
    if (WiFi.status() == WL_CONNECTED) {
        HTTPClient http;
        http.begin(serverURL);
        http.addHeader("Content-Type", "application/json");
        
        // Tạo JSON data
        StaticJsonDocument<200> doc;
        doc["device_id"] = deviceID;
        doc["heart_rate"] = readHeartRate();
        doc["temperature"] = readTemperature();
        doc["oxygen_saturation"] = readOxygenSaturation();
        doc["bp_systolic"] = readBPSystolic();
        doc["bp_diastolic"] = readBPDiastolic();
        doc["respiratory_rate"] = readRespiratoryRate();
        doc["room_temperature"] = readRoomTemperature();
        doc["humidity"] = readHumidity();
        
        String jsonString;
        serializeJson(doc, jsonString);
        
        int httpResponseCode = http.POST(jsonString);
        
        if (httpResponseCode > 0) {
            String response = http.getString();
            Serial.println("Response: " + response);
        } else {
            Serial.println("Error sending data");
        }
        
        http.end();
    }
    
    delay(30000); // Gửi dữ liệu mỗi 30 giây
}

// Các hàm đọc sensor (cần implement theo sensor thực tế)
float readHeartRate() {
    // Implement heart rate sensor reading
    return 75.0; // Giá trị mẫu
}

float readTemperature() {
    // Implement temperature sensor reading
    return 37.0; // Giá trị mẫu
}

float readOxygenSaturation() {
    // Implement oxygen saturation sensor reading
    return 98.0; // Giá trị mẫu
}

float readBPSystolic() {
    // Implement blood pressure sensor reading
    return 120.0; // Giá trị mẫu
}

float readBPDiastolic() {
    // Implement blood pressure sensor reading
    return 80.0; // Giá trị mẫu
}

float readRespiratoryRate() {
    // Implement respiratory rate sensor reading
    return 16.0; // Giá trị mẫu
}

float readRoomTemperature() {
    // Implement room temperature sensor reading
    return 24.0; // Giá trị mẫu
}

float readHumidity() {
    // Implement humidity sensor reading
    return 60.0; // Giá trị mẫu
}
